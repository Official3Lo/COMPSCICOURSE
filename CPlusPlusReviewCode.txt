#include<iostream>

char* removeBlank(char a[], int remove) {
	for (int i = 0; i < a[i]; i++) {
		while (a[i] != '\0') {
			if (a[i] == ' ') {
				remove = i;
				int d = remove;
				while (a[d] != '\0') {
					a[d] = a[d+i];
					d++;
				}
			}
			i++;
		}
	}
	return a;
}
char *getCopyof(char a[], int length)
{
	char* c = new char[length];
	for (int i = 0; i < length; i++) {
		c[i] = a[i];
	}
	return c;
}

int main() {

{
	int s = 0;

void solution(double a, double b, double c, double &s1, double &s2)
{
	cout.setf(ios::fixed);
	cout.precision(2);
	double d = b * b - 4 * a * c;
	if (d < 0)
		cout << "there is no real solution \n";
	else if (d == 0) {
		s1 = (-b) / (2 * a);
		s2 = (-b) / (2 * a);
	}
	else {
		s1 = (-b + sqrt(b * b - 4 * a * c)) / (2*a);
		s2 = (-b - sqrt(b * b - 4 * a * c)) / (2*a);
	}
}

int main()
{
	double a, b, c, x1, x2;
	cout << "please enter the coefficients of a quatratic equation:\n";

	solution(a, b, c, x1, x2);
